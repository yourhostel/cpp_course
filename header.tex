\definecolor{inlinecodefg}{HTML}{1B5E20}

\newcommand{\inlinecodetext}[1]{%
  \begingroup
  \ttfamily
  \hyphenchar\font=-1
  \exhyphenpenalty=10000
  \hyphenpenalty=10000
  \textcolor{inlinecodefg}{%
    \StrSubstitute{#1}{\string\{}{\char123}[\tmpA]%
    \StrSubstitute{\tmpA}{\string\}}{\char125}[\tmpB]%
    \StrSubstitute{\tmpB}{\string\$}{\char36}[\tmpC]%
    \StrSubstitute{\tmpC}{; }{;\allowbreak\space}[\tmpD]%
    \tmpD
  }%
  \endgroup
}

\let\oldtexttt\texttt
\renewcommand{\texttt}[1]{%
  \inlinecodetext{#1}%
}
